#### أ- من الشِّعرِ

> المصدر: https://dorar.net/alakhlaq/3729

#### - قال حسَّانُ بنُ ثابتٍ يُعَيِّرُ الحارثَ بنَ هِشامٍ بفِرارِه يومَ بَدرٍ:

  
إنْ كُنتِ كاذبةَ الذي حدَّثْتِني  
فنجوتِ منجى الحارِثِ بنِ هِشامِ  
تركَ الأحبَّةَ لم يقاتِلْ دونَهم  
ونجا برأسِ طِمِرَّةٍ [1] ولجامِ  
ملأَتْ به الفرجَينِ فارمَدَّتْ [2] به  
وثوى [3] أحبَّتُه بشَرِّ مَقامِ [4]   

#### - قال النُّويريُّ: (ومِن أبلَغِ ما قيل في الجُبْنِ من الشِّعرِ القديمِ:

  

#### قولُ الشَّاعِرِ:

  
ولو أنَّها عصفورةٌ لحسِبْتَها  
مسوَّمةً [5] تدعو عبيدًا وأزنمَا [6]   

#### ومِثلُه قولُ عُروةَ بنِ الوردِ:

  
وأشجَعَ قد أدركْتُهم فوجَدْتُهم  
يخافون خَطفَ الطَّيرِ من كُلِّ جانبِ  

#### - وقال آخَرُ:

  
ما زلتَ تحسَبُ كُلَّ شيءٍ بَعدَهم  
خيلًا تَكِرُّ [7] عليهمُ ورِجالَا  

#### - وقولُ أبي تمَّامٍ:

  
موكَلٌ بيفاعِ [8] الأرضِ يَذرَعُها  
من خِفَّةِ الخَوفِ لا من خِفَّةِ الطَّرَبِ  

#### - وقال ابنُ الرُّومي:

  
وفارِسٍ أجبَنَ من صِفرِدٍ [9]   
يحولُ أو يغورُ مِن صَفْرَهْ  
لو صاح في اللَّيلِ به صائحٌ  
لكانت الأرضُ له طَفْرَه [10]   
يرحمُه الرَّحمنُ من جُبنِه  
فيُرزَقُ الجندُ به النُّصْرَهْ) [11] .  

#### - وقال:

  
وإذا ما خلا الجبانُ بأرضٍ  
طَلَب الطَّعنَ وَحدَه والنِّزالا [12]   

#### - وقال:

  
فلو أنَّ الحياةَ تبقى لحيٍّ  
لعَدَدْنا أضَلَّنا الشُّجْعانا  
وإذا لم يكُنْ من الموتِ بُدٌّ  
فمن العَجزِ أن تكونَ جبانَا [13]   

#### - وقال آخَرُ:

  
جهلًا علينا وجُبنًا عن عَدُوِّكم  
لبِئسَت الخَلَّتانِ الجَهلُ والجُبْنُ [14]   

#### - وقال العَرجيُّ:

  
مِثلُ الضَّفادعِ نقَّاقون وَحدَهم  
إذا خَلَوا، وإذا لاقَيتَهم خُرسُ [15] !  

#### - وقيل في رجُلٍ أكولٍ جبانٍ:

  
إذا صوَّتَ العُصفورُ طار فؤادُه  
وليثٌ حديدُ النَّابِ عِندَ الثَّرائِدِ [16]   

#### - وكانرَضِيَ الله عنه يتمَثَّلُ بهذين البيتينِ كثيرًا:

  
أكان الجبانُ يرى أنَّه  
سيُقتَلُ قبلَ انقِضاءِ الأجَلْ  
فقد تُدرِكُ الحادثاتُ الجَبانَ   
ويَسلَمُ منها الشُّجاعُ البَطَلْ [17]

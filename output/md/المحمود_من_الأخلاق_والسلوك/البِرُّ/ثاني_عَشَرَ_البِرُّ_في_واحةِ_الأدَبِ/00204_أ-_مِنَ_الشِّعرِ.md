#### أ- مِنَ الشِّعرِ

> المصدر: https://dorar.net/alakhlaq/450

1- قال ابنُ الأعرابيِّ:  
ليس بما ليس به بأسٌ باسُ   
ولا يَضيرُ البِرَّ ما قال النَّاسُ [1]   
2- قال امرُؤُ القيسِ:  
اللهُ أنجَحُ ما طلَبْتُ به  
والبِرُّ خيرُ حقيبةِ الرَّحْلِ [2]   
3- وقال لَبيدٌ:  
وما البِرُّ إلَّا مُضْمَراتٌ من التُّقى  
وما المالُ إلَّا مُعْمَراتٌ ودائِعُ [3]   
4- وقال سابِقٌ البَربَريُّ:  
إنَّ التُّقى خيرُ زادٍ أنت حامِلُه   
والبِرُّ أفضَلُ شيءٍ ناله بَشَرُ [4]   
5- وقال آخَرُ:  
والإثمُ من شَرِّ ما يُصالُ به [5]   
والبِرُّ كالغَيثِ نَبتُه أَمِرُ [6]   
6- وقال آخَرُ:  
مَن لم يُنِلْك البِرُّ في حياتِه  
لم تَبْكِ عيناك على وَفاتِه [7]   
7- أنشدَ الكُرَيزيُّ:  
مِن خَيرِ ما حُزْتَه وُدٌّ لذي كَرَمٍ   
يجزيك ما عِشْتَ بالإحسانِ إحسانَا  
تلقى بشاشَتَه في قُربِه وإذا  
أنال نالَك منه البِرُّ ما كانا [8]   
8- قال الشَّاعِرُ:  
وكم صاحِبٍ أكرَمْتُه غيرَ طائِعٍ   
ولا مُكرَهٍ إلَّا لأمرٍ تعَمَّدَا  
وما كان ذاك البِرُّ إلَّا لغيرِه  
كما نَصَبوا للطَّيرِ بالحَبِّ مِصْيَدَا [9]   
9- وقال أبو العتاهيةِ:  
وإنَّ امرَأً لم يَرتَجِ النَّاسُ نَفْعَه  
ولم يأمَنوا منه الأذى لَلَئيمُ  
وإنَّ امرَأً لم يجعَلِ البِرَّ كَنْزَه  
ولو كانت الدُّنيا له لعَديمُ [10]   
10- قال الصَّرصَريُّ:  
واغرِسْ أُصولَ البِرِّ تَجْنِ ثمارَها  
فالبِرُّ أزكى مَنبَتًا للغارِسِ [11]   
11- وقال سُفيانُ بنُ عُيَينةَ:  
أبُنَيَّ إنَّ البِرَّ شَيءٌ هَيِّنُ   
وَجهٌ طَليقٌ وكَلامٌ لَيِّنُ [12]   
12- وقال آخَرُ:  
بُنَيَّ إنَّ البِرَّ شَيءٌ هَيِّنُ   
المَنطِقُ اللَّيِّنُ والطُّعَيِّمُ [13]   
13- وأنشَد أبو الحَسَنِ الهاشِميُّ:  
النَّاسُ كُلُّهم عِيالُ  
اللهِ تحتَ ظِلالِه  
فأحَبُّهم طُرًّا [14] إليه   
أبَرُّهم لعيالِه [15]   
14- قال أبو إسحاقَ الإلبِيريُّ:  
فلا تَرْضَ المعايِبَ فهي عارٌ  
عظيمٌ يُورِثُ الإنسانَ مَقْتَا  
وتهوي بالوَجيهِ مِن الثُّرَيَّا [16]   
وتُبدِلُه مكانَ الفَوقِ تَحْتَا  
كما الطَّاعاتُ تُنعِلُك الدَّراري [17]   
وتجعَلُك القريبَ وإن بَعُدْتا  
وتنشُرُ عنك في الدُّنيا جميلًا   
فتُلفي البِرَّ فيها حيثُ كُنْتا   
وتمشي في مناكِبِها كريمًا  
وتجني الحَمدَ ممَّا قد غَرَسْتا [18]   
15- قال الوليدُ بنُ يزيدَ:  
من يتَّقِ اللهَ يَجِدْ غِبَّ التُّقى   
يومَ الحسابِ صائِرًا إلى الهُدى  
إن التُّقى أفضَلُ شيءٍ في العَمَلْ  
أرى جِماعَ البِرِّ فيه قد دَخَلْ [19]   
16- قال أنَسُ بنُ زُنَيمٍ الدِّيليُّ:  
وما حمَلَتْ من ناقةٍ فَوقَ رَحْلِها   
أبَرَّ وأوفى ذِمَّةً مِن محمَّدِ [20]   
17- وقال أحمد شوقي في وَصفِ النَّبيِّ صلَّى اللهُ عليه وسلَّم:  
والبِرُّ عندك ذِمَّةٌ وفريضةٌ  
 لا مِنَّةٌ ممنونةٌ وجِباءُ [21]
